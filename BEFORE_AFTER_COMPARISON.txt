import { campaignAPI } from '../services/api';

const CampaignTable = () => {
  const [campaigns, setCampaigns] = useState([]);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState('');

  useEffect(() => {
    loadCampaigns();
  }, []);

  const loadCampaigns = async () => {
    try {
      setIsLoading(true);
      const data = await campaignAPI.getAll();  // ✅ Real API call!
      setCampaigns(data?.data || []);
    } catch (err) {
      setError('Failed to load campaigns: ' + err.message);
    } finally {
      setIsLoading(false);
    }
  };

  return (
    {isLoading && <LoadingSpinner />}
    {error && <ErrorAlert message={error} />}
    {campaigns.map(campaign => (
      <TableRow
        name={campaign.name}
        status={campaign.status}
        budget={campaign.budget}
        startDate={campaign.start_date}
        endDate={campaign.end_date}
      />
    ))}
  );
};

Improvements:
  ✅ Real API integration (/api/campaigns)
  ✅ Live database data
  ✅ Loading states with spinner
  ✅ Error handling with user feedback
  ✅ Dynamic table rendering
  ✅ Real status values (Active, Draft, Completed, etc.)
  ✅ Actual budget tracking
  ✅ Real start/end dates
  ✅ Can add/edit campaigns (with modal)
  ✅ True campaign ROI tracking
