━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Invoice Card displayed:
❌ 🏢 Unknown
   Invoice
   INV-1765189862957-3

But the data was available in database:
✅ Company ID 6 = "Innovation Labs"
✅ Invoice 4 → Client ID 6
✅ Database JOIN working properly

Root Cause:
The backend API was NOT returning the "company_name" field that the 
frontend was looking for. It was using "client_name" instead, but the 
frontend expected "company_name".

API Response Structure (BEFORE):
{
  "id": 4,
  "invoice_number": "INV-1765189862957-3",
  "client_id": 6,
  "client_name": "Innovation Labs",    ← Backend used this
  "company_name": undefined             ← Frontend looked for this
}

                          THE SOLUTION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Fixed 2 API endpoints to return BOTH company_name AND client_name:

✅ GET /api/invoices (list all invoices)
   Modified SQL query to include:
   SELECT i.*, 
          c.company_name,              ← Added this
          c.company_name AS client_name,
          c.email AS company_email,
          c.phone AS company_phone
   FROM invoices i 
   LEFT JOIN companies c ON i.client_id = c.id

✅ GET /api/invoices/:id (get single invoice)
   Modified SQL query to include:
   SELECT i.*, 
          c.company_name,              ← Added this
          c.company_name AS client_name
   FROM invoices i 
   LEFT JOIN companies c ON i.client_id = c.id

Result:
Now BOTH field names are available in the response, maintaining backward 
compatibility while fixing the UI issue.
